####### TRIMMING TO A GIVEN SEQUENCE LENGHT #######
# START R and set working directory
	tmux a -t texas
	mamba activate dada2v1.16
	R 
	# sanity check
		library(dada2); packageVersion("dada2") 
	# set working directory
		setwd("~/TexasTech") # getwd() # to check
# 2. Input cutadapt-ed files
	# define path.cut
	path.cut <- file.path("~/TexasTech/cutadapt")
	# Forward and reverse fastq filenames have the format:
	cutFs <- sort(list.files(path.cut, pattern = "_R1_001.fastq.gz", full.names = TRUE))
	cutRs <- sort(list.files(path.cut, pattern = "_R2_001.fastq.gz", full.names = TRUE))
	# Extract sample names, assuming filenames have format:
	get.sample.name <- function(fname) strsplit(basename(fname), "_")[[1]][1]
	sample.names <- unname(sapply(cutFs, get.sample.name))
	head(sample.names)
# 3. Inspect read quality 
	# Please note that I obtained my plots for quality inspection on RSTUDIO with the following command:
	# STEP 1: First the forward reads
	plotQualityProfile(cutFs[1:50], aggregate = T)
	# STEP 2: Then Reverse reads
	plotQualityProfile(cutRs[1:50], aggregate = T)
	# I am not sure how to export figures from R in Terminal but you can check it out here (spanish): https://r-coder.com/exportar-graficos-r/
# 4. Filter and trim
	# STEP 1: Create directory for new files
	# Place filtered files in filtered/ subdirectory
	filtFs <- file.path(path.cut, "filtered", paste0(sample.names, "_F_filt.fastq.gz"))
	filtRs <- file.path(path.cut, "filtered", paste0(sample.names, "_R_filt.fastq.gz"))
	# STEP 2: Filter and trim Fw to 280 bp and Rv to 210 bp (enough to overlap the ~444 bp size of V4 gene), with an increased error rate = 3,3
	out <- filterAndTrim(cutFs, filtFs, cutRs, filtRs, truncLen=c(280,210),
                              maxN=0, maxEE=c(3,3), truncQ=2, rm.phix=TRUE,
                              compress=TRUE, multithread=TRUE)
	head(out)
	# STEP 3: PLooting to see how nice are your trimms 
	plotQualityProfile(filtFs[1:50], aggregate = T)
	plotQualityProfile(filtRs[1:50], aggregate = T)

###### DENOISING #######
#
# 5. Learn error rates
# Advised by Tara, I changed the randomize parameter to "randomize = TRUE"
	# STEP 1: Forward first
	errF <- learnErrors(filtFs, multithread=TRUE, randomize = TRUE)
	# output: 112659960 total bases in 402357 reads from 3 samples will be used for learning the error rates.
	# STEP 2: Reverse then
	errR <- learnErrors(filtRs, multithread=TRUE, randomize = TRUE)
	# output: 111717900 total bases in 531990 reads from 6 samples will be used for learning the error rates.
	# STEP 3: PLot your errors
	plotErrors(errF, nominalQ=TRUE)
	plotErrors(errR, nominalQ=TRUE)
# 6. Dereplication
	# FORWARD
	derepFs <- derepFastq(filtFs, verbose=TRUE)
	# REVERSE
	derepRs <- derepFastq(filtRs, verbose=TRUE)
	# Name the derep-class objects by the sample names
	names(derepFs) <- sample.names
	names(derepRs) <- sample.names
# 7. Sample inference: DADA IT!
	# FORWARD
	dadaFs <- dada(derepFs, err=errF, multithread=TRUE)
	# REVERSE
	dadaRs <- dada(derepRs, err=errR, multithread=TRUE)
	# Then, INSPECT!
	dadaFs[[1]]
	dadaRs[[1]]
# 8. Merge pair reads 
	mergers <- mergePairs(dadaFs, derepFs, dadaRs, derepRs, verbose=TRUE)
	# Inspect the merger data.frame from the first sample
	head(mergers[[1]])
# 9. Construct sequence table
	# STEP 1
	seqtab <- makeSequenceTable(mergers)
	# STEP 2
	dim(seqtab)
	# STEP 3: Inspect distribution of sequence lengths
	table(nchar(getSequences(seqtab)))
## 9.1. Shorting sequence table to 402-431 bp size
	# Getting sequences only 402 to 431 bp size
	seqtab2 <- seqtab[,nchar(colnames(seqtab)) %in% 402:431]
	# Double-cheking
	dim(seqtab2)
	# I missed (8706-7977 = 729) sequences that were out of size range allowed for V4 gene =252bp
	table(nchar(getSequences(seqtab2)))
# 10. Remove chimera sequences
# I am not completely sure what they are but we need to remove them
# We are starting with 52780 sequences as inputs, lets see how many of them are quimeras
	seqtab.nochim2 <- removeBimeraDenovo(seqtab2, method="consensus", multithread=TRUE, verbose=TRUE)
	# Identified 33063 bimeras out of 52780 input sequences
	# This will tell us how many non-chimera sequences remain
	dim(seqtab.nochim2)
	# 19717 sequences remained
	# this formula will tell us what is the percentage of non-chimera sequences that remain,
	# So we can not the percentage of chimera sequences we removed
	sum(seqtab.nochim2)/sum(seqtab2)
	# [1] 0.9430187
# 11. Finally check how many reads have you lost in the whole process, starting with the without-primers input reads (STEP 6) to the final chimera removal (STEP 14)
	getN <- function(x) sum(getUniques(x))
	track <- cbind(out, sapply(dadaFs, getN), sapply(dadaRs, getN), sapply(mergers, getN), rowSums(seqtab.nochim2))
	# If processing a single sample, remove the sapply calls: e.g. replace sapply(dadaFs, getN) with getN(dadaFs)
	colnames(track) <- c("input", "filtered", "denoisedF", "denoisedR", "merged", "nonchim")
	rownames(track) <- sample.names
	head(track) 
# 12. Save in your working directory your "track" table as a summary table in a "tsv format" that Excel can read
	write.table(track, "read-count-tracking.tsv", quote=FALSE, sep="\t", col.names=NA)
	
####### ASSING TAXONOMY #######
# 
# 13. ASSIGN TAXONOMY: Now you will give a name to your ASVs
# For this step you will need to have 2 databases: SILVA genus (~/Rumen_16S/16S_database/silva_nr99_v138.1_train_set.fa.gz) and SILVA species (~/Rumen_16S/16S_database/silva_species_assignment_v138.1.fa.gz)
# Look this issue for more info on which database you should use: https://github.com/benjjneb/dada2/issues/1319
# You can download both databases in the most updated version from: https://zenodo.org/record/4587955#.YpZOOGiZPIU
	# STEP 1: Assign until genus level
	taxa <- assignTaxonomy(seqtab.nochim2, "~/TexasTech/silva_nr99_v138.1_train_set.fa.gz", multithread=TRUE)
	# STEP 2: Assign until species level
	taxa <- addSpecies(taxa, "~/TexasTech/silva_species_assignment_v138.1.fa.gz")
# MODIFIED ASSIGN TAXONOMY ON 01/16/26 IN LOCAL COMPUTER: Re-run with new SILVA database 138.2, launched on July 11, 2024
	# Download SILVA 138.2 database from: https://zenodo.org/records/14169026
	# Download: silva_nr99_v138.2_toGenus_trainset.fa.gz & silva_v138.2_assignSpecies.fa.gz
	# Place the databases in a new folder which I will use as output folder for this re-analysis:
		/Users/gerardodiaz/Desktop/Gerardo/UMN/Side_projects/TexasTech_resistome/Microbiome_SILVA138_2

		setwd("/Users/gerardodiaz/Desktop/Gerardo/UMN/Side_projects/TexasTech_resistome")

		# You should see [1] ‘1.34.0’
	# Then:
	# STEP 1: load "seqtab.nochim2" that was saved as "otu_table.rds" 
		seqtab.nochim2 <- readRDS("otu_table.rds")
	# STEP 2: Assign until genus level
		taxa <- assignTaxonomy(seqtab.nochim2, "~/Microbiome_SILVA138_2/silva_nr99_v138.2_toGenus_trainset.fa.gz", multithread=TRUE)
	# NOTE I did not complete species Assignment because it ate all my local 16GB RAM and couldnt finish species level classification


####### EXPORT OUTPUT #######
#
# 14. Extracting data from dada2 analysis. You may have: FASTA FILE (with the 16S sequences of each ASV), a count table (ASV table) and a taxonomy table (Tax Table)
	# giving our seq headers more manageable names (ASV_1, ASV_2...)
	asv_seqs <- colnames(seqtab.nochim2)
	asv_headers <- vector(dim(seqtab.nochim2)[2], mode="character")
	for (i in 1:dim(seqtab.nochim2)[2]) {
	asv_headers[i] <- paste(">ASV", i, sep="_")
	}
	# making and writing out a fasta of our final ASV seqs:
	asv_fasta <- c(rbind(asv_headers, asv_seqs))
	write(asv_fasta, "ASVs.fa")
	# count table (ASV table):
	asv_tab <- t(seqtab.nochim2)
	row.names(asv_tab) <- sub(">", "", asv_headers)
	write.table(asv_tab, "ASVs_counts.tsv", sep="\t", quote=F, col.names=NA)
	# tax table:
	write.table(taxa, "taxa.tsv", sep = "\t", quote=F, col.names=NA)
	#SAVING RDS OBJECTS (RDS) AND WORKSPACE (RData)
	saveRDS(seqtab.nochim2, file = "otu_table.rds")
	saveRDS(taxa, file = "taxa_table.rds")
	save.image(file = "TexasTech.RData")

        # Use awk to search for the header that starts with the keyword (e.g., >contig-100)

        if [[ -s "${fa_file%.fa}_${header_keyword}.fa" ]]; then

        else
            echo "No matching sequence found in $fa_file for header starting with '$header_keyword'."

        fi

        echo "File $fa_file not found."

#!/bin/bash

# Folder containing input BAMs
input_folder="/scratch.global/fermx014/data/diazo005/AMRplusplus/output_texastech/Alignment/BAM_files/Standard"

# Output folder
output_folder="results"
mkdir -p "$output_folder"

# Number of threads
threads=20

# Temporary list of flagstat files
flagstat_files=()

# ----------------------------
# Step 1-4: Process BAM files
# ----------------------------
for bam in "$input_folder"/*.bam; do
    base=$(basename "$bam" .bam)
    echo "Processing $base..."

    # Step 1: fixmate
    fixmate_bam="$output_folder/${base}_fixmate.bam"
    samtools fixmate -@ $threads -m "$bam" "$fixmate_bam"

    # Step 2: sort by position (multithreaded)
    positionsort_bam="$output_folder/${base}_positionsort.bam"
    samtools sort -@ $threads -o "$positionsort_bam" "$fixmate_bam"

    # Step 3: mark duplicates (multithreaded)
    markdup_bam="$output_folder/${base}_markdup.bam"
    samtools markdup -@ $threads "$positionsort_bam" "$markdup_bam"

    # Step 4: flagstat
    flagstat_txt="$output_folder/${base}_flagstat.txt"
    samtools flagstat -@ $threads "$markdup_bam" > "$flagstat_txt"

    flagstat_files+=("$flagstat_txt")

    echo "Finished $base"
done

# ----------------------------
# Step 5: Combine flagstat into TSV
# ----------------------------
# Output TSV
combined_tsv="$output_folder/combined_flagstat.tsv"

# Header row (matching the lines you want from flagstat)
echo -e "Sample\tin_total\tprimary\tsecondary\tsupplementary\tduplicates\tprimary_duplicates\tmapped\tprimary_mapped\tpaired_in_sequencing\tread1\tread2\tproperly_paired\twith_itself_and_mate_mapped\tsingletons\twith_mate_mapped_to_diff_chr\twith_mate_mapped_to_diff_chr_mapQ5" > "$combined_tsv"

# Loop over flagstat files
for f in "${flagstat_files[@]}"; do
    sample=$(basename "$f" _flagstat.txt)
    # Extract first numbers from each line and put in one row
    # Only first number before " + "
    counts=$(awk '{print $1}' "$f" | paste -sd "\t" -)
    echo -e "${sample}\t${counts}" >> "$combined_tsv"
done

echo "All flagstat files combined into $combined_tsv"
